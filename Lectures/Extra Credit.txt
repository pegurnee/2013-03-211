/**
 * Extra Credit, create a method to return the proper day of the week
 * @author Eddie Gurnee
 * @version 11/20/2013
 */
public int dayOfWeek(String day) {
	int theDay = 0;
	switch (day) {
		case "Sunday":
			theDay = 1;
			break;
		case "Monday":
			theDay = 2;
			break;
		case "Tuesday":
			theDay = 3;
			break;
		case "Wednesday":
			theDay = 4;
			break;
		case "Thursday":
			theDay = 5;
			break;
		case "Friday":
			theDay = 6;
			break;
		case "Saturday":
			theDay = 7;
			break;
		default:
			theDay = -1;
			break;
	}
	return theDay;
}
